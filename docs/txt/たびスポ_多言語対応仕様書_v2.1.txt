4.2 DeepL API 実装

// src/lib/deepl.ts

import * as deepl from &apos;deepl-node&apos;



const translator = new deepl.Translator(process.env.DEEPL_API_KEY!)



export async function translateSpotPR(japaneseText: string) {

  const targets: deepl.TargetLanguageCode[] = [&apos;en-US&apos;, &apos;zh&apos;, &apos;zh-TW&apos; as any, &apos;ko&apos;]

  const keys = [&apos;pr_en&apos;, &apos;pr_zh_cn&apos;, &apos;pr_zh_tw&apos;, &apos;pr_ko&apos;]



  const results = await Promise.all(

    targets.map(lang =&gt;

      translator.translateText(japaneseText, &apos;ja&apos;, lang)

    )

  )



  return Object.fromEntries(

    keys.map((key, i) =&gt; [key, results[i].text])

  )

}



// Webhook (承認時) で呼び出す

const translations = await translateSpotPR(spot.pr)

await supabase.from(&quot;spots&quot;).update(translations).eq(&quot;id&quot;, spotId)

// app/[locale]/spots/[id]/page.tsx の generateMetadata

export async function generateMetadata({ params }) {

  const { locale, id } = params

  const spot = await getSpot(id)

  const baseUrl = &apos;https://tabispo.app&apos;



  return {

    alternates: {

      languages: {

        &apos;ja&apos;:    `${baseUrl}/ja/spots/${id}`,

        &apos;en&apos;:    `${baseUrl}/en/spots/${id}`,

        &apos;zh-CN&apos;: `${baseUrl}/zh-CN/spots/${id}`,

        &apos;zh-TW&apos;: `${baseUrl}/zh-TW/spots/${id}`,

        &apos;ko&apos;:    `${baseUrl}/ko/spots/${id}`,

        &apos;x-default&apos;: `${baseUrl}/spots/${id}`,  // 日本語をデフォルトに

      }

    },

    title: locale === &quot;ja&quot; ? spot.seo_title || spot.name

         : locale === &quot;en&quot; ? `${spot.name} | TabiSpo`

         : `${spot.name} | たびスポ`,

    description: locale === &quot;ja&quot; ? spot.meta_description || spot.pr

               : locale === &quot;en&quot; ? spot.pr_en || spot.pr

               : locale === &quot;zh-CN&quot; ? spot.pr_zh_cn || spot.pr

               : locale === &quot;zh-TW&quot; ? spot.pr_zh_tw || spot.pr

               : spot.pr_ko || spot.pr,

  }

}
